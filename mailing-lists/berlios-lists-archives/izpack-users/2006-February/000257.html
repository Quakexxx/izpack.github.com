Anyway, it works exactly as createForPack
E.g:
<job name="Clear the Client installation">
  <executeForPack name="Client"/>
  <executeForPack name="ClientBatchs"/>
  <os family="windows" />
  <executefile name="$INSTALL_PATH\XXXX_batch\WinClear.bat" />
 </job>
